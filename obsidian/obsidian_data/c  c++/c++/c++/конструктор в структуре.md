конструктор с параметрами:
```cpp
struct point
{
private:
    int x, y;

public:
    point(int x, int y) // конструктор
    {
        this->x = x;
        this->y = y;
    }

    int get_x() { return x; }
    int get_y() { return y; }
};

int main()
{
    point pt(2, 3);
    cout << pt.get_x() << ' ' << pt.get_y() << endl;
}
...
2 3
```
