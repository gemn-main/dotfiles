```cpp
const int lolik = 123;
int main()
{
    int bobik = 234;
    int kakashka = 12;

    auto r = [=]() 
    {
        cout << lolik << endl;
        cout << bobik << endl;
        cout << kakashka << endl;
    };
    r();

    return 0;
}
123
234
12
```
`=` - все локальные переменные копируются в функцию и не имеют связь с оригиналом
