```cpp
int modul(int x)
{
    cout << "modul(int)" << endl;
    return (x > 0) ? x : -x;
}

double modul(double x)
{
    cout << "modul(double)" << endl;
    return (x > 0) ? x : -x;
}

int main()
{
    int res1 = modul(-0.3);
    double res2 = modul(-0.3);
    int res3 = modul(-3);
    double res4 = modul(-3);
    cout << "res1 = " << res1 << "   res2 = " << res2 << 
    "   res3 = " << res3 << "   res4 = " << res4 << endl;
    return 0;
}
```
