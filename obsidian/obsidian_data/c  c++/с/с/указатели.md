```c
int main(void)
{
    char d = 22;
    char *gpt = &d;
    printf("gtp = %p, *gpt = %d, d = %d\n" ,gpt, *gpt, d);

    *gpt = 12;
    printf("gtp = %p, *gpt = %d, d = %d\n" ,gpt, *gpt, d);
    return 0;
}
...
gpt = 0062ff03, *gpt = 22, d = 22
gpt = 0062ff03, *gpt = 12, d = 12
```
