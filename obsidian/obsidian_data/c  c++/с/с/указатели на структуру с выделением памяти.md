```c
struct tag_vector
{
    double x, y;
};

int main(void)
{
    struct tag_vector *ptr_v = malloc(sizeof(struct tag_vector));
    ptr_v->x = 12.12;
    ptr_v->y = 13.13;

    printf("x = %f, y = %f", ptr_v->x, ptr_v->y);
    free(ptr_v);
}
```
