чтобы изменить последовательность надо установить seed `srand(numbers(int))`

```c
int main(void){
	srand(12);
    int res1 = rand();
    int res2 = rand();
    printf("res1 = %d, res2 = %d\nrand: %d, %d, %d", res1, res2, rand(), rand(), rand());
 
    return 0;
}
...
res1 = 77, res2 = 5628
rand: 1558, 29052, 6232
```

чтобы seed менялся со временем надо библиотеку time; 
```c
#include <time.h>
int main(void)
{
    time_t t = time(NULL);
    srand(t);
    int res1 = rand();
    int res2 = rand();
    printf("res1 = %d, res2 = %d\nrand: %d, %d, %d\nt = %d", res1, res2, rand(), rand(), rand(), t);
 
    return 0;
}
res1 = 13017, res2 = 10594
rand: 28525, 9102, 29370
t = 1731332579
	```


создание диапазона:
```c
int main(void)
{
    srand(time(NULL));
    int range = 15;
    int res1 = rand() % range;
    int res2 = rand() % range; 
    printf("res1 = %d, res2 = %d", res1, res2);
 
    return 0;
}
```

генерация вещественных чисел:
```c

int main(void)
{
    srand(time(NULL));
    int range = 15;
    int res1 = rand() % range;
    int res2 = rand() % range; 
    printf("res1 = %d, res2 = %d, float = %f", res1, res2,                  (double)rand()/ (double)RAND_MAX);
 
    return 0;
}
...
res1 = 11, res2 = 14, float = 0.880490
```
