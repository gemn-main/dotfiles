void swap_2(int* a, int* b)
{
    int t = *a;
    *a = *b;
    *b = t;
}

const char* find_spase(const char* arr)
{
    while(*arr != '\0'){
        if (*arr == ' ')
            return arr;
        *arr++;
    }
}

int main()
{
    int x = 12, y = 1;
    printf("x = %d, y = %d\n", x, y);
    swap_2(&x, &y);
    printf("x = %d, y = %d\n", x, y);
    char str[] = "sdf sd";
    const  char *res = find_spase(str);
    printf("res = %p", res);
    return 0;
}
