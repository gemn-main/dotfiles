    int indx = 1;
    for (int i = indx; i < MARKSS-1; i++){
        printf("%d = %d\n", marks[i], marks[i+1]);   
        marks[i] = marks[i+1];
    }

    for (int j = 0; j<MARKSS; j++)
        printf("%d, ", marks[j]);
    return 0;
}
