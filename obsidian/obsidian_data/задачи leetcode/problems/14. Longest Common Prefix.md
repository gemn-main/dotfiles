```python
class Solution(object):
    def longestCommonPrefix(self, strs):
        min_str = min(strs)
        o = ''
        for i in range(1, len(min_str)+1):           
            if len(min_str[:i]) < len(o):
                break
            flak = True
            for y in range(len(strs)):
                if min_str[:i] not in strs[y][:i]:
                    flak = False
            if flak == True:
                o = min_str[:i]
        return o 
```
