class Solution(object):
    def findWords(self, words):
        o = []
        m = []
        stringi= ["qwertyuiop", "asdfghjkl", "zxcvbnm"]
        for i in range(len(words)):
            for k in range(3):
                if words[i][0].lower() in stringi[k]:
                    o.append(k)
        for i in range(len(words)):
            b = True
            for j in range(len(words[i])):
                if b == False:
                    break
                for k in range(3):
                    if words[i][j] in stringi[k]:
                        if k != o[i]:
                            b = False
                if j == (len(words[i])-1) and b != False:
                    m.append(words[i])
        return m
