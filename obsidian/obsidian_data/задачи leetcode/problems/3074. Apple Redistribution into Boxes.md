```python
class Solution(object):
    def minimumBoxes(self, apple, capacity):
        capacity.sort(reverse=True)
        sumapp =  sum(apple)
        for i in range(len(capacity)):
            sumapp -= capacity[i]
            if sumapp <= 0:
                return i+1
```
