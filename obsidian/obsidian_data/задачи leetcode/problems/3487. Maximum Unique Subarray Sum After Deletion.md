class Solution {
public:
    int maxSum(vector<int>& nums) {
        vector<int> vec;
        for (auto i : nums){
            if (count(vec.begin(), vec.end(), i) == 0){
                if (i >= 1)
                    vec.push_back(i);
                }
        }
        int sum = 0;
        for (size_t i = 0; i < vec.size(); ++i) {
            sum += vec[i];
            }
        if (vec.size()== 0){
            sum += *max_element(nums.begin(), nums.end());
        }
        return sum;
        
    }
};
