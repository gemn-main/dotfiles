(breadth-first search - BFS)
```cpp
queue<int> q;
vector<int> adj[100];
bool visited[100];
int dist[100];

void search(int x)
{
    visited[x] = true;
    dist[x] = 0;
    q.push(x);
    while (!q.empty())
    {
        int s = q.front();
        cout << "обработка вершины: " << s << '\n';
        q.pop();
        for (auto u : adj[s])
        {
            if (!visited[u])
            {
                visited[u] = true;
                dist[u] = dist[s] + 1;
                q.push(u);
            }
        }
    }
}

int main()
{
    adj[1].push_back(2);
    adj[1].push_back(3);
    adj[2].push_back(3);
    adj[3].push_back(4);
    adj[3].push_back(5);
    search(1);
    return 0;
}
```
