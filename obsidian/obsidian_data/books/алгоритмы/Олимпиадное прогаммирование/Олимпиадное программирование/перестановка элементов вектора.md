```cpp
#include <bits/stdc++.h>
#include <fstream>
using namespace std;
using std::ios;

vector<int> permutation;
int n = 3;
vector<bool> chosen(n + 1, false);
void search()
{
    if (permutation.size() == n)
    {
        cout << "( ";
        for (int i : permutation)
            cout << i << ' ';
        cout << ")" << '\n';
    }
    else
    {
        for (int i = 1; i <= n; i++)
        {
            if (chosen[i])
                continue;
            chosen[i] = true;
            permutation.push_back(i);
            search();
            chosen[i] = false;
            permutation.pop_back();
        }
    }
}

int main()
{
    search();
    return 0;
}
```
