(depth-first search - DFS)
исследует каждую ветвь до конца 
```cpp
vector<int> adj[100];
bool visited[100];

int loli(int u)
{
    cout << "Посещаем: " << u << endl;
    visited[u] = true;
    for (int b : adj[u])
        if (!visited[b])
            loli(b);
}

int main()
{
    adj[1].push_back(2);
    adj[1].push_back(3);
    adj[2].push_back(3);
    adj[3].push_back(4);
    adj[3].push_back(5);
    loli(1);
    return 0;
}
```

`adj[100];` - список смежности
`visited[100]` - история посещения 

*временная сложность 0(n + m)*
n - число вершин
m - число ребер
