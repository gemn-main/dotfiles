каждая вершина содержит пару (b, w)
b - на следующую вершину 
w - вес
```cpp
int main()
{
    vector<pair<int, int>> adj[5];
    adj[1].push_back({2, 5});
    adj[2].push_back({3, 7});
    adj[2].push_back({4, 6});
    adj[3].push_back({4, 5});
    adj[4].push_back({1, 2});
    for (int u = 1; u <= 4; ++u)
    {
        for (auto edge : adj[u])
        {
            cout << u << " -> " << edge.first << ", вес: " << edge.second << endl;
        }
    }
}
```
