В этом случая каждой вершине x сопоставляется список смежности, включающий вершины, соединенные с ребром x 

```cpp
int main()
{
    vector<int> adj[5];
    adj[1].push_back(2);
    adj[2].push_back(3);
    adj[2].push_back(4);
    adj[3].push_back(4);
    adj[4].push_back(1);
    for (int i = 1; i <= 4; ++i)
    { // Перебираем вершины от 1 до 4
        cout << i << " -> ";
        for (int neighbor : adj[i])
        { // Перебираем всех соседей вершины i
            cout << neighbor << " ";
        }
        cout << endl;
    }
}
```
