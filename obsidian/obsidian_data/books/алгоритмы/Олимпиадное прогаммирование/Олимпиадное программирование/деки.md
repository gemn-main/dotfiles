в деках в отличие от векторов можно добавлять элементы как в начало так и в конец
```cpp
int main()
{
    deque<int> d;
    d.push_back(1);   // [1]
    d.push_back(3);   // [1, 3]
    d.push_front(12); // [12, 1, 3]
    d.pop_back();     //[12, 1]
    d.pop_front();    // [1]
    for (auto x : d)
        cout << x << ' ';
    return 0;
}
```
