int const n = 6;
vector<tuple<int, int, int>> edges;
int distancee[n];

int main()
{
    for (int i = 1; i <= n; i++)
    {
        distancee[i] = 1e9;
    }
    edges.push_back({1, 2, 2});
    edges.push_back({1, 3, 3});
    edges.push_back({3, 4, -2});
    edges.push_back({2, 4, 3});
    edges.push_back({1, 4, 7});
    edges.push_back({4, 5, 2});
    edges.push_back({2, 5, 5});

    distancee[1] = 0;
    for (int i = 1; i <= n - 1; i++)
    {
        for (auto e : edges)
        {
            int a, b, w;
            tie(a, b, w) = e;
            distancee[b] = min(distancee[b], distancee[a] + w);
        }
    }
    int b = 0;
    for (auto u : distancee)
    {
        if (b != 0)
            cout << "до вершины " << b << ": " << u << '\n';
        b++;
    }
}
